# ---------------------------------------------------------
# CREATE FOLDERS + PROJECTS
# ---------------------------------------------------------

foreach ($Folder in $Folders)
{
    $FullPath = Join-Path $PWD $Folder

    New-Item -ItemType Directory -Path $FullPath -Force | Out-Null

    $ProjectName = ($Folder -replace '[\\/]', '.') -replace '^src\.',''

    Push-Location $FullPath

    # Create class library
    dotnet new classlib -n $ProjectName --framework net9.0

    # Add to solution
    dotnet sln "C:\_D\DotNetEngineeringLab\DotNetEngineeringLab.slnx" add "$ProjectName\$ProjectName.csproj"

    Pop-Location
}
